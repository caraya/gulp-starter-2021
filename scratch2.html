If you’re using Gulp and Dust.js the code may look like this:

const gulp = require('gulp');
const dust = require('gulp-dust');

gulp.task('default', () =>
  gulp
    .src('templates/list.html')
    .pipe(dust())
    .pipe(gulp.dest('dist'))
);

And if you’re using Nunjucks the Gulp task looks like this:

const gulp = require('gulp');
const nunjucks = require('gulp-nunjucks');

gulp.task('default', () =>
  gulp
    .src('templates/greeting.html')
    .pipe(nunjucks.compile({ name: 'Sindre' }))
    .pipe(gulp.dest('dist'))
);
